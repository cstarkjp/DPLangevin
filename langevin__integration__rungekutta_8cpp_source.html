50void BaseLangevin::rk_f4_and_stochastic(
51 const dbl_vec_t &density_grid_aux_old,
52 const dbl_vec_t &k1,
53 const dbl_vec_t &k2,
54 const dbl_vec_t &k3,
55 rng_t &rng
56)
57{
58 mean_density = 0.0;
59 for (auto i=0; i<n_cells; i++)
60 {
61 auto k4 = nonlinear_rhs(i, density_grid_aux_old);
62 density_grid[i] += dts*(k1[i] + 2*(k2[i] + k3[i]) + k4);
63 #if !APPROXIMATE_POISSON_DISTBN
64 poisson_rng = int_poisson_dist_t(lambda_product*density_grid[i]);
65 gamma_rng = dbl_gamma_dist_t(poisson_rng(rng), 1.0);
66 #else
67 double mu = lambda_product * density_grid[i];
68 if (mu > MU_THRESHOLD)
69 {
70 normal = dbl_normal_dist_t(mu, sqrt(mu));
71 gamma_rng = dbl_gamma_dist_t(normal(rng), 1.0);
72 }
73 else
74 {
75 poisson_rng = int_poisson_dist_t(lambda_product*density_grid[i]);
76 gamma_rng = dbl_gamma_dist_t(poisson_rng(rng), 1.0);
77 }
78 #endif
79 density_grid[i] = gamma_rng(rng)/lambda;
80 mean_density += density_grid[i];
81 }
82 mean_density /= static_cast<double>(n_cells);
83}
