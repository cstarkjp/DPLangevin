40double DPLangevin::nonlinear_rhs(const int i_cell, const dbl_vec_t &field) const
41{
42 // Non-linear term, which is quadratic in the DP Langevin
43 const double quadratic_term
44 = -quadratic_coeff*field[i_cell]*field[i_cell];
45
46 // Integration of diffusion
47 double diffusion_sum = 0.0;
48 int n_neighbors = neighbors[i_cell].size();
49 for (auto i=0; i<n_neighbors; i++)
50 {
51 auto i_neighbor = neighbors[i_cell][i];
52 diffusion_sum += field[i_neighbor];
53 }
54 diffusion_sum = D*(diffusion_sum - n_neighbors*field[i_cell]);
55 return diffusion_sum + quadratic_term;
56}
