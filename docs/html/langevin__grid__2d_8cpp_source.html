16 neighbors = std::vector<int_vec_t>(n_x*n_y, int_vec_t(4));
17
18 switch (parameters.grid_topology)
19 {
20 case GridTopology::PERIODIC:
21 // Periodic grid topology in both x and y
22 for (auto y=0; y<n_y; y++)
23 {
24 for (auto x=0; x<n_x; x++)
25 {
26 i_cell = x + y*n_x;
27 i_up = (y < n_y-1) ? i_cell + n_x : x;
28 i_down = (y > 0) ? i_cell - n_x : x + (n_y-1)*n_x;
29 i_right = (x < n_x-1) ? i_cell + 1 : 0 + y*n_x;
30 i_left = (x > 0) ? i_cell - 1 : n_x-1 + y*n_x;
31
32 neighbors[i_cell][0] = i_up; // Up
33 neighbors[i_cell][1] = i_down; // Down
34 neighbors[i_cell][2] = i_right; // Right (VMB: left)
35 neighbors[i_cell][3] = i_left; // Left (VMB: right)
36 }
37 }
38 return true;
39
40 case GridTopology::BOUNDED:
41 // Bounded grid topology in both x and y
42 for (auto y=1; y<n_y-1; y++)
43 {
44 for (auto x=1; x<n_x-1; x++)
45 {
46 // i_cell is the index of the flattened grid
47 i_cell = x + y*n_x;

54 }
