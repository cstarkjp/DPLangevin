22bool SimDP::initialize_grid()
23{
24 int i_cell;
25 switch (p.initial_condition)
26 {
27 case (InitialCondition::RANDOM_GAUSSIAN):
28 dpLangevin->ic_random_uniform(*rng);
29 return true;
30 case (InitialCondition::CONSTANT_VALUE):
31 dpLangevin->ic_constant_value(p.aux_values.at(0));
32 return true;
33 case (InitialCondition::SINGLE_SEED):
34 if (p.grid_dimension==GridDimension::D1)
35 {
36 i_cell = ( static_cast<int>(p.aux_values.at(1)) );
37 if (i_cell<0 or i_cell>=p.n_x) { return false; }
38 }
39 else if (p.grid_dimension==GridDimension::D2)
40 {
41 i_cell = (static_cast<int>(p.aux_values.at(1))
42 + static_cast<int>(p.aux_values.at(2))*p.n_x);
43 if (i_cell<0 or i_cell>=p.n_x*p.n_y) { return false; }
44 }
45 else { return false; }
46 dpLangevin->ic_single_seed(i_cell, p.aux_values.at(0));
47 return true;
48 case (InitialCondition::RANDOM_UNIFORM):
49 dpLangevin->ic_random_uniform(*rng);
50 return true;
51 default:
52 return false;
53 }
54}
