8bool Langevin::construct_1D_grid(const Parameters parameters)
9{
10 const int n_x = parameters.n_x;
11 neighbors = std::vector<int_vec_t>(n_x, int_vec_t(2));
12
13 // Everywhere except the grid ends
14 for (auto i=1; i<n_x-1; i++)
15 {
16 // Each cell has a L and R neighbor whose indexes are specified here
17 neighbors[i][0] = i-1;
18 neighbors[i][1] = i+1;
19 }
20
21 // Grid ends
22 switch (parameters.grid_topology)
23 {
24 case GridTopology::PERIODIC:
25 // Each end cell neighbor is the other end cell, so wrap the indexes
26 neighbors[0][0] = n_x-1; // left-end left
27 neighbors[0][1] = 1; // left-end right
28 neighbors[n_x-1][0] = n_x-2; // right-end left VMB: [n_x-1][0] = n_x-2;
29 neighbors[n_x-1][1] = 0; // right-end right
30
31 case GridTopology::BOUNDED:
32 // Link each end cell to its adjacent cell only
33 neighbors[0] = int_vec_t(1, 1);
34 neighbors[n_x-1] = int_vec_t(1, n_x-2);
35
36 default:
37 return false;
38 }
39}
