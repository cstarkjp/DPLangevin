45bool SimDP::initialize()
46{
47 if (not dpLangevin->construct_grid(p)) {
48 std::cout
49 << "SimDP::initialize failure: couldn't construct grid"
50 << std::endl;
51 return false;
52 }
53 if (not dpLangevin->initialize_grid(p, *rng)) {
54 std::cout
55 << "SimDP::initialize failure: couldn't initialize grid"
56 << std::endl;
57 return false;
58 }
59 dpLangevin->set_coefficients(coefficients);
60 n_epochs = count_epochs();
61 t_epochs = dbl_vec_t(n_epochs, 0.0);
62 mean_densities = dbl_vec_t(n_epochs, 0.0);
63 // Treat epoch#0 as the initial grid state
64 // So after initialization, we are nominally at epoch#1
65 i_next_epoch = 1;
66 t_next_epoch = p.dt;
67 if (not dpLangevin->check_boundary_conditions(p))
68 {
69 std::cout << "Failure: wrong number of boundary conditions" << std::endl;
70 return false;
71 }
72 if (not choose_integrator())
73 {
74 std::cout << "Failure: unable to choose integrator" << std::endl;
75 return false;
76 }
77 is_initialized = true;
78 return is_initialized;
79}
