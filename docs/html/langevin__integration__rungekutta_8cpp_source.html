8void Langevin::integrate_rungekutta(rng_t &rng)
9{
10 // Runge-Kutta integration of the non-linear term and diffusion.
11 // Update of cells is done in the same loop as last RK step
12 // for efficiency
13 rk_f1(aux_cell_old, k1);
14 rk_f2f3(aux_cell_old, aux_cell_new, k2, dtm);
15 // Swap contents is O(1), better than old = fast
16 aux_cell_old.swap(aux_cell_new);
17 rk_f2f3(aux_cell_old, aux_cell_new, k3, dt);
18 aux_cell_old.swap(aux_cell_new);
19 rk_f4_and_stochastic(aux_cell_old, k1, k2, k3, rng);
20}

47// RK fourth function and stochastic step, all in the same loop
48void Langevin::rk_f4_and_stochastic(
49 const dbl_vec_t &aux_old,
50 const dbl_vec_t &k1,
51 const dbl_vec_t &k2,
52 const dbl_vec_t &k3,
53 rng_t &rng
54)
55{
56 mean_density = 0.0;
57 for (auto i=0; i<n_cells; i++)
58 {
59 auto k4 = nonlinear_rhs(i, aux_old);
60 cell_density[i] += dts*(k1[i] + 2*(k2[i] + k3[i]) + k4);
61 #if !APPROXIMATE_POISSON_DISTBN
62 poisson = int_poisson_dist_t(lambda_product*cell_density[i]);
63 gamma = dbl_gamma_dist_t(poisson(rng), 1.0);
64 #else
65 double mu = lambda_product * cell_density[i];
66 if (mu > MU_THRESHOLD)
67 {
68 normal = dbl_normal_dist_t(mu, sqrt(mu));
69 gamma = dbl_gamma_dist_t(normal(rng), 1.0);
70 }
71 else
72 {
73 poisson = int_poisson_dist_t(lambda_product*cell_density[i]);
74 gamma = dbl_gamma_dist_t(poisson(rng), 1.0);
75 }
76 #endif
77 cell_density[i] = gamma(rng)/lambda;
78 mean_density += cell_density[i];
79 }
80 mean_density /= static_cast<double>(n_cells);
81}
